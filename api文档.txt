使用示例
var obj=this.tweens.add({
    targets: image,

    //要渐变达到的属性值
    //透明度渐变成1,x,y到哪
    x,y:
    alpha: 1,
    //
    ease:
    //两秒后开始
    delay: 2000,
    //一共用6秒
    duration: 6000
    //-1为重复
    repeat: -1,
    //
    repeatDelay: 1000,
    //到达效果维持多长时间，开始继续下一次循环 ，与yoyo对应
    hold: 1000
    //是否返回
    yoyo: true,
    //设置初始暂停
    paused: true
});
//停止动画
obj.stop();
obj.play();
